# Quick Start

Build a complete Business-as-Code application in 5 minutes using sdk.do.

## What We'll Build

A customer management system with:

- Customer database with semantic types
- AI-powered customer insights
- Event-driven welcome emails
- Daily reporting workflow

## Prerequisites

- Node.js 18+ installed
- 5 minutes of your time

## Step 1: Create Project (30 seconds)

```bash
mkdir customer-app
cd customer-app
pnpm init
pnpm add sdk.do schema.org.ai tsx
```

## Step 2: Configure Environment (30 seconds)

Create `.env`:

```bash
SDK_DATABASE_URL=postgres://user:pass@localhost:5432/customers
SDK_OPENAI_API_KEY=sk-your-api-key-here
SDK_AI_MODEL=gpt-5
```

## Step 3: Write Application Code (2 minutes)

Create `index.ts`:

```typescript
import $, { db, ai, on, send, every } from 'sdk.do'

// ============================================
// 1. Database Operations
// ============================================

async function createCustomer(name: string, email: string) {
  const customer = await db.create($.Customer, {
    name,
    email,
    dateCreated: new Date().toISOString(),
    customerStatus: 'Active',
  })

  console.log('âœ“ Created customer:', customer.$id)

  // Trigger welcome event
  send($.Customer.created, customer)

  return customer
}

async function listCustomers() {
  const customers = await db.list($.Customer, {
    where: { customerStatus: 'Active' },
    limit: 100,
  })

  console.log(`âœ“ Found ${customers.length} active customers`)
  return customers
}

// ============================================
// 2. AI-Powered Insights
// ============================================

async function analyzeCustomer(customer: any) {
  const insights = await ai.generate(
    `Analyze this customer and provide business insights:
     Name: ${customer.name}
     Email: ${customer.email}
     Status: ${customer.customerStatus}

     Provide: segment, value tier, engagement recommendations`,
    { schema: $.AnalysisReport }
  )

  console.log('âœ“ Generated insights:', insights)

  // Update customer with AI insights
  await db.update(customer, {
    segment: insights.segment,
    valueTier: insights.valueTier,
  })

  return insights
}

// ============================================
// 3. Event Handlers
// ============================================

// Welcome new customers
on($.Customer.created, async (customer) => {
  console.log('â†’ New customer event:', customer.name)

  // Generate personalized welcome message
  const welcomeMessage = await ai.generate(`Write a personalized welcome email for ${customer.name}`)

  // Send welcome email event
  send($.Email.send, {
    to: customer.email,
    subject: `Welcome, ${customer.name}!`,
    body: welcomeMessage,
  })

  console.log('âœ“ Sent welcome email to:', customer.email)
})

// Handle email sending
on($.Email.send, async (email) => {
  console.log('â†’ Email queued:', email.to)
  // In production, integrate with email service (SendGrid, etc.)
  console.log('âœ“ Email sent:', email.subject)
})

// ============================================
// 4. Scheduled Workflows
// ============================================

// Daily customer report
every($.Daily, async () => {
  console.log('â†’ Running daily report...')

  const customers = await listCustomers()

  // Calculate metrics
  const newToday = customers.filter((c) => {
    const created = new Date(c.dateCreated)

  })

    Total Active: ${customers.length}
    New Today: ${newToday.length}
    â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”â”
  `)
})

// ============================================
// 5. Main Application
// ============================================

async function main() {
  console.log('\nðŸš€ Starting Customer Management System...\n')

  // Create sample customers
  console.log('Creating customers...')
  const alice = await createCustomer('Alice Johnson', 'alice@example.com')
  const bob = await createCustomer('Bob Smith', 'bob@example.com')
  const carol = await createCustomer('Carol Williams', 'carol@example.com')

  console.log('\nâ”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€\n')

  // Generate AI insights
  console.log('Generating AI insights...')
  await analyzeCustomer(alice)
  await analyzeCustomer(bob)

  console.log('\nâ”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€\n')

  // List all customers
  const allCustomers = await listCustomers()

  console.log('\nâ”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€\n')

  // Batch AI processing
  console.log('Batch processing customer insights...')
  const batchResults = await ai.batch(
    allCustomers.slice(0, 3).map((customer) => ({
      prompt: `Generate a one-line insight for customer: ${customer.name}`,
    }))
  )

  batchResults.forEach((result, i) => {
    console.log(`âœ“ ${allCustomers[i].name}: ${result}`)
  })

  console.log('\nâœ… Customer Management System Ready!\n')
}

// Run the application
main().catch((error) => {
  console.error('âŒ Error:', error)
  process.exit(1)
})
```

### Add More Features

**Customer Search**:

```typescript
import { ai } from 'sdk.do'

async function searchCustomers(query: string) {
  const embedding = await ai.embed(query)

  const results = await db.list($.Customer, {
    where: {
      $semantic: {
        embedding,
        threshold: 0.8,
      },
    },
  })

  return results
}
```

**Customer Segmentation**:

```typescript
async function segmentCustomers() {
  const segments = {
    enterprise: [],
    smb: [],
    individual: [],
  }

  const customers = await db.list($.Customer)

  for (const customer of customers) {
    const segment = await ai.generate(`Classify customer segment: ${customer.name}`, { schema: { type: 'string', enum: ['enterprise', 'smb', 'individual'] } })

    segments[segment].push(customer)
  }

  return segments
}
```

## Real-World Examples

### E-commerce Order Processing

```typescript
import $, { db, ai, on, send } from 'sdk.do'

on($.Order.created, async (order) => {
  // Validate order with AI
  const validation = await ai.generate(`Validate order for fraud: ${JSON.stringify(order)}`, { schema: { isValid: 'boolean', reason: 'string' } })

  if (validation.isValid) {
    send($.Order.approved, order)
  } else {
    send($.Order.flagged, { ...order, reason: validation.reason })
  }
})

on($.Order.approved, async (order) => {
  // Process payment
  send($.Payment.process, { orderId: order.$id, amount: order.total })

  // Update inventory
  for (const item of order.items) {
    const product = await db.get($.Product, item.productId)
    await db.update(product, {
      inventory: product.inventory - item.quantity,
    })
  }

  // Send confirmation
  send($.Email.send, {
    to: order.customerEmail,
    template: 'order-confirmation',
    data: order,
  })
})
```

### Content Publishing Workflow

```typescript
import $, { db, ai, on, every } from 'sdk.do'

on($.Article.drafted, async (article) => {
  // AI content review
  const review = await ai.generate(`Review article for quality, SEO, and readability: ${article.content}`, { schema: $.ReviewReport })

  await db.update(article, {
    reviewScore: review.score,
    suggestions: review.suggestions,
  })

  if (review.score >= 8) {
    send($.Article.readyForPublish, article)
  } else {
    send($.Article.needsRevision, article)
  }
})

every($.Daily, async () => {
  // Auto-publish scheduled articles
  const scheduled = await db.list($.Article, {
    where: {
      status: 'scheduled',
      publishDate: { $lte: new Date() },
    },
  })

  for (const article of scheduled) {
    await db.update(article, { status: 'published' })
    send($.Article.published, article)
  }
})
```
