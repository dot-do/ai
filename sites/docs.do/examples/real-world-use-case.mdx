# Real-World Use Case

Complete documentation system for a multi-product SaaS platform.

## Scenario

A SaaS company with multiple products needs:

- Unified documentation across all products
- Version management for each product release
- Multi-language support
- Analytics on documentation usage
- AI-powered search and recommendations

## Implementation

```typescript
import $, { docs, ai, every } from 'docs.do'

class DocumentationSystem {
  private products = ['sdk', 'database', 'events', 'workflows']
  private languages = ['en', 'es', 'fr', 'de', 'ja']

  async initialize() {
    // Index all product documentation
    for (const product of this.products) {
      await this.indexProduct(product)
    }

    // Schedule daily updates
    every($.Daily, () => this.updateDocumentation())

    // Enable analytics tracking
    this.enableAnalytics()
  }

  async indexProduct(product: string) {
    // Index from multiple sources
    await docs.indexDomain(`${product}.do`, {
      includeAPI: true,
      includeExamples: true,
      includeGuides: true,
    })

    // Generate embeddings for semantic search
    const allDocs = await docs.list($.TechArticle, {
      where: { domain: `${product}.do` },
    })

    for (const doc of allDocs) {
      const embedding = await ai.embed(doc.articleBody)
      await docs.update(doc, { embedding })
    }

    // Translate to all languages
    for (const lang of this.languages) {
      if (lang !== 'en') {
        await docs.translateDomain(`${product}.do`, lang)
      }
    }
  }

  async search(query: string, options: SearchOptions) {
    // Hybrid search (semantic + keyword)
    const results = await docs.search(query, {
      semantic: true,
      hybridWeight: 0.6,
      minScore: 0.7,
      ...options,
    })

    // Track search
    await docs.track('search', {
      query,
      resultsCount: results.length,
      userId: options.userId,
    })

    return results
  }

  async updateDocumentation() {
    // Find outdated docs
    const outdated = await docs.list($.TechArticle, {
      where: {
        dateModified: {
          $lt: new Date(Date.now() - 90 * 24 * 60 * 60 * 1000),
        },
      },
    })

    // Queue for review
    for (const doc of outdated) {
      await this.queueReview(doc)
    }

    // Sync translations
    for (const doc of outdated) {
      await docs.syncTranslations(doc.$id)
    }
  }

  async enableAnalytics() {
    // Track all page views
    app.use(async (req, res, next) => {
      if (req.path.startsWith('/docs/')) {
        await docs.track('page_view', {
          documentId: extractDocId(req.path),
          userId: req.user?.$id,
          referrer: req.headers.referer,
        })
      }
      next()
    })
  }

  async getRecommendations(docId: string, userId: string) {
    // Get user's reading history
    const history = await docs.userHistory(userId)

    // Find similar docs
    const doc = await docs.get(docId)
    const similar = await docs.findSimilar(doc.embedding, {
      limit: 5,
      excludeSelf: true,
    })

    // Get popular in same domain
    const popular = await docs.popular({
      domain: doc.domain,
      limit: 5,
    })

    return { similar, popular, history }
  }
}

// Initialize system
const docsSystem = new DocumentationSystem()
await docsSystem.initialize()
```
