# API Reference

Complete API documentation for docs.do.

## Core Functions

### docs.create()

Create a new documentation entry.

```typescript
function create<T>(type: T, data: Partial<T>, options?: CreateOptions): Promise<T>
```

**Parameters:**

- `type` - Document type ($.TechArticle, $.APIReference, $.HowTo)
- `data` - Document data
- `options` - Optional creation options

**Example:**

```typescript
const doc = await docs.create($.TechArticle, {
  headline: 'Getting Started',
  articleBody: '...',
  domain: 'sdk.do',
})
```

### docs.search()

Search documentation with multiple strategies.

```typescript
function search(query: string, options?: SearchOptions): Promise<SearchResult[]>
```

**Parameters:**

- `query` - Search query
- `options` - Search options (domains, type, limit, semantic, etc.)

**Example:**

```typescript
const results = await docs.search('database operations', {
  domains: ['database.do'],
  semantic: true,
  limit: 20,
})
```

### docs.get()

Get a document by ID.

```typescript
function get(id: string, options?: GetOptions): Promise<Document>
```

**Parameters:**

- `id` - Document ID
- `options` - Get options (version, language)

**Example:**

```typescript
const doc = await docs.get('sdk.do/getting-started', {
  version: '2.0.0',
  language: 'es',
})
```

### docs.update()

Update an existing document.

```typescript
function update(idOrDoc: string | Document, updates: Partial<Document>, options?: UpdateOptions): Promise<Document>
```

**Parameters:**

- `idOrDoc` - Document ID or document object
- `updates` - Fields to update
- `options` - Update options (version, changeLog)

**Example:**

```typescript
await docs.update(doc, {
  articleBody: 'Updated content',
  version: '2.0.0',
})
```

### docs.delete()

Delete a document.

```typescript
function delete(idOrDoc: string | Document): Promise<void>
```

### docs.list()

List documents by type.

```typescript
function list<T>(type: T, options?: ListOptions): Promise<T[]>
```

**Example:**

```typescript
const articles = await docs.list($.TechArticle, {
  where: { domain: 'sdk.do' },
  limit: 50,
})
```

### docs.relate()

Create relationships between documents.

```typescript
function relate(from: Document, predicate: Predicate, to: Document): Promise<void>
```

**Example:**

```typescript
await docs.relate(guide, $.relatedTo, apiReference)
```

### docs.related()

Query related documents.

```typescript
function related(doc: Document, predicate: Predicate, type?: Type): Promise<Document[]>
```

### docs.versions()

Get all versions of a document.

```typescript
function versions(id: string): Promise<Version[]>
```

### docs.translate()

Translate a document.

```typescript
function translate(id: string, targetLanguage: string, options?: TranslateOptions): Promise<Document>
```

### docs.track()

Track analytics events.

```typescript
function track(event: string, data: Record<string, any>): Promise<void>
```

### docs.analytics()

Get analytics data.

```typescript
function analytics(idOrQuery: string, options?: AnalyticsOptions): Promise<Analytics>
```
