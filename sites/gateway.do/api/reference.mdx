# API Reference

## gateway.get()

Make GET request.

```typescript
function get(url: string, options?: RequestOptions): Promise<any>
```

### Example

```typescript
const data = await gateway.get('https://api.example.com/users', {
  auth: { type: 'bearer', token: process.env.TOKEN },
  cache: { enabled: true, ttl: 3600 },
})
```

## gateway.post()

Make POST request.

```typescript
function post(url: string, options?: RequestOptions): Promise<any>
```

### Example

```typescript
const user = await gateway.post('https://api.example.com/users', {
  data: { name: 'John', email: 'john@example.com' },
})
```

## RequestOptions

```typescript
interface RequestOptions {
  // Authentication
  auth?: AuthOptions

  // Data
  data?: any
  params?: Record<string, any>
  headers?: Record<string, string>

  // Caching
  cache?: CacheOptions

  // Rate limiting
  rateLimit?: RateLimitOptions

  // Transformation
  transform?: {
    request?: (data: any) => any
    response?: (data: any) => any
  }

  // Retry
  retry?: RetryOptions

  // Timeout
  timeout?: number
}
```

## AuthOptions

```typescript
interface AuthOptions {
  type: 'bearer' | 'basic' | 'apiKey' | 'oauth2' | 'custom'
  token?: string
  username?: string
  password?: string
  key?: string
  value?: string
  in?: 'header' | 'query'
}
```

## CacheOptions

```typescript
interface CacheOptions {
  enabled: boolean
  ttl?: number
  key?: string
  tags?: string[]
}
```

## RateLimitOptions

```typescript
interface RateLimitOptions {
  requests: number
  window: number
  strategy?: 'fixed-window' | 'sliding-window'
}
```

## RetryOptions

```typescript
interface RetryOptions {
  max: number
  delay?: number
  backoff?: 'fixed' | 'exponential'
  circuitBreaker?: CircuitBreakerOptions
}
```

## gateway.config()

Configure gateway.

```typescript
function config(options: ConfigOptions): void
```

### Example

```typescript
gateway.config({
  timeout: 30000,
  cache: { enabled: true, ttl: 3600 },
  rateLimit: { requests: 100, window: 60000 },
})
```

## gateway.analytics

Get usage analytics.

```typescript
const analytics = await gateway.analytics.usage({
  period: 'last_7_days',
})
```
