## Quick Actions

<AgentActions>
  <Action
    label="Start Video Call"
    icon="video"
    onClick={async () => {
      const session = await agent.create({
        $type: $.AgentSession,
        agent: 'tom',
        mode: 'video_call',
        capabilities: ['screen_share', 'code_review', 'pair_programming']
      })

      return session
    }}

/>

  <Action
    label="Schedule Meeting"
    icon="calendar"
    onClick={async () => {
      const meeting = await db.create($.Meeting, {
        agent: 'tom',
        attendees: [$.user],
        duration: 30,
        type: 'code_review'
      })

      await send($.Meeting.scheduled, meeting)
      return meeting
    }}

/>

  <Action
    label="Send Code for Review"
    icon="code"
    onClick={async () => {
      // Open file picker
      const files = await selectFiles()

      const review = await agent.execute({
        agent: 'tom',
        task: 'code_review',
        files: files,
        priority: 'high'
      })

      return review
    }}

/>

</AgentActions>
