# Basic Development Utilities Usage

Simple, practical examples of using dev.do.

## Example 1: Simple Operation

```typescript
import { dev } from 'dev.do'

// Basic usage
const result = await dev({
  data: 'example input',
})

console.log('Result:', result)
```

## Example 2: With Options

```typescript
import { dev } from 'dev.do'

// With configuration options
const result = await dev({
  data: 'example input',
  options: {
    format: 'json',
    validate: true,
  },
})

console.log('Configured result:', result)
```

## Example 3: Error Handling

```typescript
import { dev, DevError } from 'dev.do'

async function safeOperation(data: any) {
  try {
    const result = await dev({
      data,
      options: { strict: true },
    })
    return { success: true, result }
  } catch (error) {
    if (error instanceof DevError) {
      console.error('Development Utilities error:', error.message)
      return { success: false, error: error.message }
    }
    throw error
  }
}

const result = await safeOperation('input data')
```

## Example 4: Batch Processing

```typescript
import { dev } from 'dev.do'

// Process multiple items
const items = ['item1', 'item2', 'item3']

const results = await Promise.all(items.map((item) => dev({ data: item })))

console.log('Batch results:', results)
```

## Example 5: With TypeScript

```typescript
import { dev } from 'dev.do'
import type { DevOptions, DevResult } from 'dev.do'

// Type-safe operation
const options: DevOptions = {
  data: 'typed input',
  validate: true,
  format: 'json',
}

const result: DevResult = await dev(options)
console.log('Typed result:', result)
```

## Complete Example

```typescript
import { dev } from 'dev.do'

async function main() {
  // Initialize
  console.log('Starting dev operation...')

  // Process data
  const result = await dev({
    data: 'example data',
    options: {
      format: 'json',
      validate: true,
      prettify: true,
    },
  })

  // Handle result
  if (result.success) {
    console.log('Success:', result.data)
  } else {
    console.error('Failed:', result.error)
  }

  return result
}

// Run
main().catch(console.error)
```
