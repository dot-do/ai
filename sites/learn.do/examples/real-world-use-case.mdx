# Real-World Use Case

## Enterprise Training Platform

```typescript
import { learn } from 'learn.do'

class EnterpriseTraining {
  async initialize() {
    // Create learning paths
    const paths = await this.createPaths()

    // Onboard employees
    await this.onboardEmployees()

    // Track and report
    await this.setupReporting()
  }

  async createPaths() {
    return [
      await learn.createPath({
        name: 'Developer Onboarding',
        courses: ['platform-intro', 'sdk-basics', 'advanced-topics'],
      }),
      await learn.createPath({
        name: 'Leadership Track',
        courses: ['management-basics', 'team-building'],
      }),
    ]
  }

  async onboardEmployees() {
    const employees = await getNewEmployees()

    for (const employee of employees) {
      const role = employee.role
      const path = await learn.getPathForRole(role)
      await learn.enroll(employee, path)
    }
  }

  async setupReporting() {
    every($.Weekly, async () => {
      const report = await learn.generateCompanyReport({
        metrics: ['completion', 'engagement', 'certifications'],
      })

      await sendToManagement(report)
    })
  }
}
```
