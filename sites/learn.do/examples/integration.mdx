# Integration Example

## API Integration

```typescript
import { learn } from 'learn.do'
import express from 'express'

const app = express()

app.post('/api/enroll', async (req, res) => {
  const { userId, courseId } = req.body
  await learn.enroll(userId, courseId)
  res.json({ success: true })
})

app.get('/api/progress/:userId', async (req, res) => {
  const progress = await learn.getUserProgress(req.params.userId)
  res.json({ progress })
})
```
