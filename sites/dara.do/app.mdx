        $type: $.AgentSession,
        agent: 'dara',

        },

    }}

## Live Business Metrics

export const LiveMetrics = async () => {
  const metrics = await db.get($.BusinessMetrics, { period: 'current' })

const metricCards = [
{
name: 'MRR',
value: metrics.mrr,
change: metrics.mrr_change,
trend: metrics.mrr_trend,
icon: 'trending-up',
},
{
name: 'Burn Rate',
value: metrics.burn_rate,
change: metrics.burn_rate_change,
trend: metrics.burn_rate_trend,
icon: 'fire',
},
{
name: 'Cash Runway',
value: `${metrics.runway_months} months`,
alert: metrics.runway_months < 12,
icon: 'clock',
},
{
name: 'CAC Payback',
value: `${metrics.cac_payback_months} months`,
change: metrics.cac_payback_change,
icon: 'dollar-sign',
},
]

return (

<div className="grid grid-cols-2 md:grid-cols-4 gap-4">
  {metricCards.map((metric) => (
    <MetricCard
      key={metric.name}
      {...metric}
      onDrillDown={async () => {
        // Start voice call with Dara to discuss this metric
        await send($.AgentCall.requested, {
          agent: 'dara',
          topic: `Discuss ${metric.name}`,
          context: metric,
        })
      }}
    />
  ))}
</div>
) }

<LiveMetrics />

## Schedule Recurring Check-ins

export const RecurringMeetings = () => {
  const meetings = [
    {
      name: 'Weekly Business Review',
      frequency: '0 9 * * 1', // Monday 9am
      duration: 30,
      topics: ['Metrics review', 'Alerts', 'Recommendations'],
    },
    {
      name: 'Monthly Financial Close',
      frequency: '0 10 1 * *', // 1st of month, 10am
      duration: 60,
      topics: ['Month-end close', 'Forecast update', 'Budget variance'],
    },
    {
      name: 'Quarterly Strategic Planning',
      frequency: '0 14 1 1,4,7,10 *', // First day of quarter, 2pm
      duration: 120,
      topics: ['OKR review', 'Strategic initiatives', 'Resource planning'],
    },
  ]

return (

<div className="space-y-3">
  {meetings.map((meeting) => (
    <RecurringMeetingCard
      key={meeting.name}
      meeting={meeting}
      onSchedule={async () => {
        await every(meeting.frequency, async () => {
          await send($.Meeting.scheduled, {
            agent: 'dara',
            name: meeting.name,
            duration: meeting.duration,
            topics: meeting.topics,
            attendees: [$.user],
          })
        })
      }}
    />
  ))}
</div>
) }

<RecurringMeetings />
