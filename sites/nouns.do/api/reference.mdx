# nouns.do API Reference

Complete API reference for working with semantic entity types.

## Core Types

### Thing

Base type for all Schema.org entities.

```typescript
interface Thing {
  $id?: string
  $type: string
  name?: string
  description?: string
  url?: string
  image?: string | ImageObject
  identifier?: string
  alternateName?: string
  sameAs?: string
}
```

### Person

Represents an individual person.

```typescript
interface Person extends Thing {
  $type: 'Person'
  givenName?: string
  familyName?: string
  additionalName?: string
  email?: string
  telephone?: string
  jobTitle?: string
  worksFor?: Organization
  address?: PostalAddress
  birthDate?: string
  nationality?: string
}
```

### Organization

Represents a company or organization.

```typescript
interface Organization extends Thing {
  $type: 'Organization'
  legalName?: string
  taxID?: string
  industry?: string
  foundingDate?: string
  numberOfEmployees?: number
  address?: PostalAddress
  founder?: Person
  parentOrganization?: Organization
  subOrganization?: Organization[]
}
```

### Product

Represents a product or service.

```typescript
interface Product extends Thing {
  $type: 'Product'
  brand?: string | Brand
  model?: string
  sku?: string
  gtin?: string
  price?: number
  priceCurrency?: string
  availability?: 'InStock' | 'OutOfStock' | 'PreOrder'
  category?: string
  manufacturer?: Organization
  offers?: Offer
}
```

### Place

Represents a physical location.

```typescript
interface Place extends Thing {
  $type: 'Place'
  address?: PostalAddress
  geo?: GeoCoordinates
  telephone?: string
  openingHours?: string
  containedInPlace?: Place
  hasMap?: string
}
```

## $ Semantic Proxy

### Type Access

```typescript
import $ from 'sdk.do'

// Access entity types
const PersonType = $.Person
const OrgType = $.Organization
const ProductType = $.Product

// Type properties
PersonType.$type // 'Person'
PersonType.$$url // 'https://schema.org/Person'
```

### Type Methods

```typescript
// Check if type exists
$.hasType('Person') // true
$.hasType('InvalidType') // false

// Get type hierarchy
$.getParent('Person') // 'Thing'
$.getAncestors('Person') // ['Thing']
$.getSubtypes('Organization') // ['Corporation', 'LocalBusiness', ...]
```

## Validation Functions

### validate()

Validates an entity against its type schema.

```typescript
function validate(entity: any, type: TypeReference): ValidationResult

interface ValidationResult {
  valid: boolean
  errors: ValidationError[]
}

interface ValidationError {
  path: string
  message: string
  value?: any
}
```

**Example**:

```typescript
import { validate } from 'sdk.do'

const person = {
  $type: 'Person',
  name: 'John',
  email: 'invalid-email',
}

const result = validate(person, $.Person)
console.log(result.valid) // false
console.log(result.errors) // [{ path: 'email', message: 'Invalid email format' }]
```

### validateProperty()

Validates a single property value.

```typescript
function validateProperty(value: any, property: PropertyDefinition): boolean
```

**Example**:

```typescript
import { validateProperty } from 'sdk.do'

const emailValid = validateProperty('john@example.com', {
  name: 'email',
  type: 'string',
  format: 'email',
})
console.log(emailValid) // true
```

## Type Registry

### registry.getType()

Gets a type definition.

```typescript
function getType(typeName: string): TypeDefinition

interface TypeDefinition {
  $type: string
  $id: string
  label: string
  description: string
  parent: string
  properties: Property[]
  subtypes: string[]
}
```

**Example**:

```typescript
import { registry } from 'nouns.do/core'

const personType = registry.getType('Person')
console.log(personType.parent) // 'Thing'
console.log(personType.properties) // Array of property definitions
```

### registry.registerType()

Registers a custom type.

```typescript
function registerType(definition: TypeDefinition): void
```

**Example**:

```typescript
import { registry } from 'nouns.do/core'

registry.registerType({
  $type: 'Lead',
  $id: 'https://example.com/types/Lead',
  label: 'Sales Lead',
  description: 'A potential customer',
  parent: 'Person',
  properties: [
    { name: 'leadSource', type: 'string' },
    { name: 'leadScore', type: 'number' },
    { name: 'leadStatus', type: 'string', enum: ['New', 'Qualified', 'Converted', 'Lost'] },
  ],
  subtypes: [],
})
```

### registry.hasType()

Checks if a type exists.

```typescript
function hasType(typeName: string): boolean
```

**Example**:

```typescript
registry.hasType('Person') // true
registry.hasType('InvalidType') // false
```

## Schema Mapper

### toJsonLD()

Converts entity to JSON-LD format.

```typescript
function toJsonLD(entity: any): object
```

**Example**:

```typescript
import { mapper } from 'nouns.do/core'

const person = { $type: 'Person', name: 'John' }
const jsonld = mapper.toJsonLD(person)
console.log(jsonld)
// {
//   '@context': 'https://schema.org',
//   '@type': 'Person',
//   'name': 'John'
// }
```

### fromJsonLD()

Converts JSON-LD to entity format.

```typescript
function fromJsonLD(jsonld: any): object
```

**Example**:

```typescript
const jsonld = {
  '@context': 'https://schema.org',
  '@type': 'Person',
  name: 'John',
}

const entity = mapper.fromJsonLD(jsonld)
console.log(entity) // { $type: 'Person', name: 'John' }
```

## Type Guards

### isPerson()

Type guard for Person entities.

```typescript
function isPerson(entity: Thing): entity is Person {
  return entity.$type === 'Person'
}
```

### isOrganization()

Type guard for Organization entities.

```typescript
function isOrganization(entity: Thing): entity is Organization {
  return entity.$type === 'Organization'
}
```

### isProduct()

Type guard for Product entities.

```typescript
function isProduct(entity: Thing): entity is Product {
  return entity.$type === 'Product'
}
```

**Example**:

```typescript
function processEntity(entity: Thing) {
  if (isPerson(entity)) {
    console.log('Person:', entity.name, entity.email)
  } else if (isOrganization(entity)) {
    console.log('Organization:', entity.legalName || entity.name)
  } else if (isProduct(entity)) {
    console.log('Product:', entity.name, entity.price)
  }
}
```
