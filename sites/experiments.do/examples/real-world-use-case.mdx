Complete production implementation example.

## Overview

This example demonstrates a full production setup.

## Implementation

\`\`\`typescript
import \$, { experiments, db, on, every } from 'sdk.do'

// Setup
const config = {
// Configuration
}

// Run periodically
await every($.Daily, async () => {
  const result = await experiments.run(config)
  await db.create($.Result, result)
})

// Handle events
on($.Event, async (event) => {
await experiments.run({ event })
})
