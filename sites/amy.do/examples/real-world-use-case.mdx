### Phase 2: Automation

```typescript
// Automate recurring tasks
await every($.Hourly, async () => {
  const tasks = await db.list($.Task, {
    status: 'pending',
    scheduled_for: { $lte: new Date() },
  })

  for (const task of tasks) {
    await marketingAgent.execute(task)
    await db.update($.Task, task.$id, { status: 'completed' })
  }
})

// Daily reporting
await every($.Daily, async () => {
  const report = await marketingAgent.generateReport({
    period: 'yesterday',
    metrics: ['metric1', 'metric2', 'metric3'],
  })

  await db.create($.Report, report)
  await send($.Report.daily, report)
})
```
