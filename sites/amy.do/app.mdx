## Quick Actions

<AgentActions>
  <Action
    label="Marketing Strategy Call"
    icon="strategy"
    onClick={async () => {
      const session = await agent.create({
        $type: $.AgentSession,
        agent: 'amy',
        mode: 'voice_call',
        agenda: ['Q1 strategy', 'Campaign review', 'Budget allocation'],
        duration: 45
      })

      return session
    }}

/>

  <Action
    label="Content Review"
    icon="document"
    onClick={async () => {
      const files = await selectFiles()

      const review = await agent.execute({
        agent: 'amy',
        task: 'content_review',
        files: files,
        aspects: ['messaging', 'seo', 'cta', 'audience_fit']
      })

      return review
    }}

/>

  <Action
    label="Campaign Performance"
    icon="chart"
    onClick={async () => {
      const report = await agent.execute({
        agent: 'amy',
        task: 'campaign_analysis',
        period: 'last_30_days',
        metrics: ['conversions', 'roi', 'engagement', 'reach']
      })

      return report
    }}

/>

  <Action
    label="Generate Content Ideas"
    icon="lightbulb"
    onClick={async () => {
      const ideas = await agent.execute({
        agent: 'amy',
        task: 'content_ideation',
        context: {
          industry: await db.get($.Business, $.user.business_id),
          trends: await db.list($.Trend, { period: 'last_week' }),
          competitors: await db.list($.Competitor)
        }
      })

      return ideas
    }}

/>

</AgentActions>

## Active Campaigns

export const ActiveCampaigns = async () => {
  const campaigns = await db.list($.Campaign, {
    status: 'active',
    owner: 'amy'
  })

return (

<div className="space-y-4">
  {campaigns.map((campaign) => (
    <CampaignCard
      key={campaign.$id}
      campaign={campaign}
      onCallAmy={async () => {
        await send($.AgentCall.requested, {
          agent: 'amy',
          topic: `Campaign: ${campaign.name}`,
          priority: 'high',
        })
      }}
    />
  ))}
</div>
) }

<ActiveCampaigns />
